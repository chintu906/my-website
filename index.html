# AI Question Answering Tool # Install: pip install openai import openai # Enter your OpenAI API key here openai.api_key = "YOUR_API_KEY_HERE" def ask_ai(question): response = openai.ChatCompletion.create( model="gpt-4", messages=[ {"role": "system", "content": "You are a helpful assistant who can answer any question in any language."}, {"role": "user", "content": question} ], max_tokens=500, temperature=0.7 ) answer = response['choices'][0]['message']['content'] return answer # Main loop while True: user_question = input("\nAsk your question (or type 'exit' to quit): ") if user_question.lower() == 'exit': break answer = ask_ai(user_question) print("\nAI Answer:", answer)
